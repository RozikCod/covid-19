def validate_country_name(name):
    """Validate country name"""
    if not name or not name.strip():
        return False, "Country name cannot be empty"
    if len(name) < 2:
        return False, "Country name too short"
    return True, "Valid"

def validate_case_data(data):
    """Validate case data"""
    if data['deaths'] > data['confirmed']:
        return False, "Deaths cannot exceed confirmed cases"
    if data['recovered'] > data['confirmed']:
        return False, "Recovered cannot exceed confirmed cases"
    return True, "Valid"
